### 1、项目网络状态监听并跳转路由
``` ts
onMounted(() => {
  window.addEventListener('offline', ()=>{
    // 网络异常时触发
    router.push({ name: 'Disconnect' })
    sessionStorage.locationUrl=window.location.href;
  });
  window.addEventListener('online',()=>{
    // 网络恢复正常时触发
    window.location.href=sessionStorage.locationUrl
  });
})
```
